    // 优化一
class Solution {
    public int[] twoSum(int[] nums, int target) {
        // 优化版本：使用哈希表
        int[] result = new int[2];
        HashMap<Integer, Integer> hashmap = new HashMap<Integer, Integer>();
        for(int i=0; i<nums.length; i++){
            if(hashmap.containsKey(target-nums[i])) {
                result[0]=i;
                result[1]=hashmap.get(target-nums[i]);
                return result;
            }
            hashmap.put(nums[i], i);
        }
        return result;
    }
}

    2. 最接近的三数之和(Leetcode16)

    暴力求解：三重循环遍历.时间复杂度为 O(N^3)O(N3)

    优化一：排序+双指针

    优化二：针对优化一再优化，小优化

    // 优化一
class Solution {
    public int threeSumClosest(int[] nums, int target) {
        // 优化一：排序，双指针
        Arrays.sort(nums);
        int start;
        int end;
        int best=1000000;
        for(int i=0; i<nums.length-2; i++) {
            start=i+1;
            end=nums.length-1;
            while(start<end) {
                int sum=nums[i]+nums[start]+nums[end];
                if(sum==target) {
                    return target;
                }
                if(Math.abs(sum-target)<Math.abs(best-target)) {
                    best=sum;
                }
                if(sum<target) {
                    // start右移
                    start++;
                } else {
                    end--;
                }
            }
        }
        return best;
    }
}

    3. 删除有序数组中的重复项(Leetcode26)

    注意：这里只能原地修改，且将数组放在最前面，故有些思路不可行。

    原始解法：一开始想的是记录重复不重复的数组元素的位置，再将不重复的元素移动到此位置上，也就是使用了指针，但是思路较为混乱，可以考虑如下的双指针的解法。

    优化解法一：双指针。一个指向最前面，一个指向最后面。

    // slow/fast双指针
class Solution {
    public int removeDuplicates(int[] nums) {
        // 第一种双指针
        if(nums==null||nums.length==1) {
            return nums.length;
        }
        int slow=0;
        int fast=1;
        while(fast<nums.length) {
            // 判断逻辑
            if(nums[fast]!=nums[fast-1]) {
                slow++;
                nums[slow]=nums[fast];
            }
            fast++;
        }
        return slow+1;
    }
}

    5. 三数之和

    • 这一道题，首先想到的思路：排序+双指针。但是还要考虑到不能重复。一开始只有去重2，没有去重1。导致不行。这里需要考虑到所有情况，不然就会出错。另外，还需要借鉴别人的优化解法，思考巧妙之处。
    • 需要复习Java的泛型，集合知识（这里感受到了基础的重要性）。
    class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        // 排序+首尾指针(先写出基本的再做些小优化)
        List<List<Integer>> result = new ArrayList();
        Arrays.sort(nums);

        if(nums.length<3) {
            return result;
        }

        for(int i=0; i<nums.length-2; i++) {
            int left=i+1;
            int right=nums.length-1;

            // 去重1
            if(i>0&&nums[i]==nums[i-1]) {
                continue;
            }

            while(left<right) {

                // 去重2
                if(nums[left]==nums[left-1]&&left>i+1) {
                    left++;
                    continue;
                }
                if(right<nums.length-1) {
                    if(nums[right]==nums[right+1]) {
                        right--;
                        continue;
                    }
                }

                if((nums[i]+nums[left]+nums[right])>0) {
                    right--;
                } else if((nums[i]+nums[left]+nums[right])<0) {
                    left++;
                } else {
                    result.add(Arrays.asList(nums[i],nums[left],nums[right]));
                    right--;
                    left++;
                }
            }
        }
        return result;
    }
}

    6.移除链表元素(leetcode203)

    • 分析问题：这里需要删除链表中指定的元素，一开始想的是，借助循环（条件是下一个结点不为null）——> 出现问题是：如果只有一个元素那么这个下一个结点不就没法判断了吗（一种解决方法是单独判断一个结点的情况，还有一种解决方法是新建一个结点，它的下一个结点指向头结点），逐个比较，去删除元素。（接着由于新建的结点指向了头结点，故需要操作的话，需要拿到一个引用去使用）。

    • 注意，新的知识刚刚接触到，就是学习的，不要打击自己的信心。后面会了就行。

    • 解决方法一：新建结点pre，下一个为head元素

    • 解决方法二：删除头结点时另外做考虑(这里即分为几种情况去考虑，显示为null，再世头部开始的几个结点为null，派出了之后，就能按照正常的写代码了)

    • 解决方法三：递归

    // 解决方法一

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode removeElements(ListNode head, int val) {
        ListNode pre = new ListNode(-1);
        pre.next = head; // pre的下一个结点为头结点
        ListNode demo = pre;
        while(demo.next!=null) {
            if(demo.next.val==val) {
                // 删除
                demo.next=demo.next.next;
            } else {
                // 指向下一个结点
                demo=demo.next;
            }
        }
        return pre.next;
    }
}

    public ListNode removeElements(ListNode head, int val) {

                prev.next=prev.next.next;
            }else{
                prev=prev.next;
            }
        }
        return head;
    }
}

    7.旋转链表(leetcode61)

    从这题开始，你要明白，怎么开启一道题目。

    • 分析思路：通过次数，可以从尾部开始的第k个的next为null(第二步)，尾部的为开头的元素对接上(第一步）。遍历链表得到长度。
    • 以下这种方法思路对了，但是可能有些地方有些问题需要更正。思考可以改进的地方：我这里出现的问题是指向模糊了，还有没有搞清楚的问题，链表赋值没有搞懂。这里可以专门复习补充。
    • 整体思路是：计算链表长度，与移动次数k整除得到余数a，这个a就可以得到如何指向，从最后数，第a+1个指向null，它的下一个为首部。再考虑一些特殊情况：。
    // 第一版错误答案

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode rotateRight(ListNode head, int k) {
        ListNode pre = new ListNode(-1);
        pre.next = head; // 下一个节点是头部节点，指向当前首部结点
        ListNode cur = pre; // 当前结点
        /*一个困惑的地方：若使用ListNode demo = pre;是做了什么操作呢 */
        // 获得链表长度
        int length = 0;
        while(cur.next != null) {
            length++;
            cur = cur.next; // 循环结束，cur为最后一个元素
        }
        // 整除
        if (head==null) {
            return head;
        }
        int a = k%length;
        // 考虑几种特殊情况
        // 1. 链表个数为0
        // 2. a为0整除了 (包括了length=1的情形)
        if (a==0) {
            return head;
        }
        // 正常流程
        cur=pre;
        for(int i=0; i<(length-a); i++) {
            cur=cur.next;
        }
        cur.next.next=head;
        pre.next=cur.next.next;
        cur.next=null;
        return pre.next;
    }
}

    3. 当前结点上，一个cur，一个difNode去比较是否相等。这两个当前结点是需要操作的结点。可以直接跳转到difNode。这个可以确定在哪个节点上。

    解题思路二：递归。

    反思：

    • 这里出现问题的地方还是，使用哪个结点不清晰，混乱了，先可以手动模拟，熟练了脑子中模拟，
    • 先是多用些结点，后面再优化，一步一步来吗。
    • 注意各个结点之间变化的关系，多用就熟练了。

    参考链接：https://leetcode-cn.com/problems/remove-duplicates-from-sorted-list-ii/solution/fu-xue-ming-zhu-di-gui-die-dai-yi-pian-t-wy0h/

    /*解题思路一*/

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode deleteDuplicates(ListNode head) {
        ListNode dummyNode = new ListNode(-1);
        dummyNode.next = head;
        ListNode prev = dummyNode;
        ListNode cur = head; // cur当前结点
        while (cur!=null) {
            int difNum = 0;
            ListNode difNode = cur;
            // 计算当前结点，和下面结点重复的个数,注意difNode
            while(difNode!=null&&difNode.val==cur.val) {
                difNum++;
                difNode = difNode.next;
            }
            // 对个数进行处理
            if(difNum>1) { // 有重复的,就需要跳过这一段
                prev.next = difNode;
            } else { // 无重复的,cur指向difnode
                // prev = prev.next; // 这是精髓所在，这里随着遍历，但是这里一想就写错了
                prev = cur; // 无重复的时候，pre一直指向cur就是，这里有点模糊，就做实验
            }
            cur = difNode;
        }
        return dummyNode.next;
    }
}
